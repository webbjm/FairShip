110 {

112  static FairGeoInterface *geoFace=geoLoad->getGeoInterface();

116  FairGeoMedium *ShipMedium=media->getMedium(name);
117 
118  if (!ShipMedium)

123  TGeoMedium* medium=gGeoManager->GetMedium(name);
