1387def ptrack(zlayer,ptrackhits,nrwant,window):

1390# zlayer= dictionary with z-position of planes: zlayer[iplane][0]==z-position (list of length 1 :-).
1391# ptrackhits is dictionary with hits in a projection: ptrackhits[plane number][0]=list-of-hits
1392# ndrop= nr of hits that can be missing in this projection
1393# nrwant= min nr of hits on a track in this projection
1394# window: is the size of the search window
1395
1396# get first and last plane number (needs to be consecutive, and also contains empty planes
1397# should be included in the list!
1398 planes=list(zlayer.keys())
1399 planes.sort()
1400 i_1=planes[0]
1401 i_2=planes[len(planes)-1]
1402 ndrop=len(planes)-nrwant

1410 # nhitw: wanted number of hits when looking for a track
1411 for idrop in range(ndrop):
1412 #only start if wanted nr hits <= the nr of planes
1413 nrhitmax=i_2-i_1-idrop+1
1414 if nhitw<=nrhitmax:
1415 for k in range(idrop+1):
1416 #calculate the id of the first and last plane for this try.
1417 ifirst=i_1+k
1418 ilast=i_2-(idrop-k)
1419 #now loop over hits in first/last planes, and construct line
1420 dz=zlayer[ilast][0]-zlayer[ifirst][0]

1423 #for ifr in range(len(ptrackhits[ifirst][0])):

1425 if ptrackhits[ifirst][1][ifr]==0:
1426 xfirst= ptrackhits[ifirst][0][ifr]
1427 #hits in last plane
1428 for il in range(len(ptrackhits[ilast][0])):
1429 #has this hit been used already for a track: skip
1430 if ptrackhits[ilast][1][il]==0:
1431 xlast= ptrackhits[ilast][0][il]
1432 nrhitsfound=2
1433 tancand=(xlast-xfirst)/dz
1434 #fill temporary hit list for track-candidate with -1
1435 trcand=(i_2-i_1+2)*[-1]
1436 #loop over in between planes
1437 for inbetween in range(ifirst+1,ilast):
1438 #can wanted number of hits be satisfied?
1439 if nrhitsfound+ilast-inbetween>=nhitw:
1440 #calculate candidate hit
1441 xwinlow=xfirst+tancand*(zlayer[inbetween][0]-zlayer[ifirst][0])-window
1442 for im in range(len(ptrackhits[inbetween][0])):
1443 #has this hit been used?
1444 if ptrackhits[inbetween][1][im]==0:
1445 xin= ptrackhits[inbetween][0][im]

1448 #if hit larger than upper edge of window: goto next plane

1450 break

1452 if trcand[inbetween]==-1:
1453 nrhitsfound+=1

1455 #looped over in between planes, collected enough hits?
1456 if nrhitsfound==nhitw:
1457 #mark used hits
1458 trcand[ifirst]=ifr
1459 trcand[ilast]=il
1460 for i in range(ifirst,ilast+1):
1461 if trcand[i]>=0:
1462 ptrackhits[i][1][trcand[i]]=1
1463 #store the track
1464 nrtracks+=1
1465 #print "ptrack nrtracks",nrtracks
1466 tracks[nrtracks]=trcand
1467 #print "ptrack: tracks",tracks
1468 return nrtracks,tracks
