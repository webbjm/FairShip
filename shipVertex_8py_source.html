369 def VertexError(self,t1,t2,PosDir,CovMat=None,scalFac=None):
370# with improved Vx x,y resolution
371 a,u = PosDir[t1]['position'],PosDir[t1]['direction']
372 c,v = PosDir[t2]['position'],PosDir[t2]['direction']
373 Vsq = v.Dot(v)
374 Usq = u.Dot(u)
375 UV = u.Dot(v)
376 ca = c-a
377 denom = Usq*Vsq-UV**2
378 tmp2 = Vsq*u-UV*v
379 Va = ca.Dot(tmp2)/denom
380 tmp2 = UV*u-Usq*v
381 Vb = ca.Dot(tmp2)/denom
382 X = (a+c+Va*u+Vb*v) * 0.5
383 l1 = a - X + u*Va # l2 = c - X + v*Vb
384 dist = 2. * ROOT.TMath.Sqrt( l1.Dot(l1) )
385 if not CovMat: return X,dist
386 T = ROOT.TMatrixD(3,12)
387 for i in range(3):
388 for k in range(4):
389 for j in range(3):
390 KD = 0
391 if i==j: KD = 1
392 if k==0 or k==2:
393 # cova and covc
394 temp = ( u[j]*Vsq - v[j]*UV )*u[i] + (u[j]*UV-v[j]*Usq)*v[i]
395 sign = -1
396 if k==2 : sign = +1
397 T[i][3*k+j] = 0.5*( KD + sign*temp/denom )
398 elif k==1:
399 # covu
400 aNAZ = denom*( ca[j]*Vsq-v.Dot(ca)*v[j] )
401 aZAN = ( ca.Dot(u)*Vsq-ca.Dot(v)*UV )*2*( u[j]*Vsq-v[j]*UV )
402 bNAZ = denom*( ca[j]*UV+(u.Dot(ca)*v[j]) - 2*ca.Dot(v)*u[j] )
403 bZAN = ( ca.Dot(u)*UV-ca.Dot(v)*Usq )*2*( u[j]*Vsq-v[j]*UV )
404 T[i][3*k+j] = 0.5*( Va*KD + u[i]/denom**2*(aNAZ-aZAN) + v[i]/denom**2*(bNAZ-bZAN) )
405 elif k==3:
406 # covv
407 aNAZ = denom*( 2*ca.Dot(u)*v[j] - ca.Dot(v)*u[j] - ca[j]*UV )
408 aZAN = ( ca.Dot(u)*Vsq-ca.Dot(v)*UV )*2*( v[j]*Usq-u[j]*UV )
409 bNAZ = denom*( ca.Dot(u)*u[j]-ca[j]*Usq )
410 bZAN = ( ca.Dot(u)*UV-ca.Dot(v)*Usq )*2*( v[j]*Usq-u[j]*UV )
411 T[i][3*k+j] = 0.5*(Vb*KD + u[i]/denom**2*(aNAZ-aZAN) + v[i]/denom**2*(bNAZ-bZAN) )
412 transT = ROOT.TMatrixD(12,3)
413 transT.Transpose(T)
414 CovTracks = ROOT.TMatrixD(12,12)
415 tlist = [t1,t2]
416 for k in range(2):
417 for i in range(6):
