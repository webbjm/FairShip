62def parse_histograms(filepath):
63 """
64 This function parses a file containing histograms of branching ratios.
65
66 It places them in a dictionary indexed by the decay string (e.g. 'd_K0_e'),
67 as a pair ([masses...], [branching ratios...]), where the mass is expressed
68 in GeV.
69 """
70 with open(filepath) as f:
71 lines = f.readlines()
72 # Define regular expressions matching (sub-)headers and data lines
73 th1f_exp = re.compile(r'^TH1F\|.+')
74 header_exp = re.compile(r'^TH1F\|(.+?)\|B(?:R|F)/U2(.+?)\|.+? mass \‍(GeV\‍)\|?')
75 subheader_exp = re.compile(r'^\s*?(\d+?),\s*(\d+?\.\d+?),\s*(\d+\.\d+)\s*$')
76 data_exp = re.compile(r'^\s*(\d+?)\s*,\s*(\d+\.\d+)\s*$')
77 # Locate beginning of each histogram
78 header_line_idx = [i for i in range(len(lines)) if th1f_exp.match(lines[i]) is not None]
79 # Iterate over histograms
80 histograms = {}
81 for offset in header_line_idx:
82 # Parse header
83 mh = header_exp.match(lines[offset])
84 if mh is None or len(mh.groups()) != 2:
85 raise ValueError(f"Malformed header encountered: {lines[offset]}")
86 decay_code = mh.group(1)
87 # Parse sub-header (min/max mass and number of points)
88 ms = subheader_exp.match(lines[offset+1])
89 if ms is None or len(ms.groups()) != 3:
90 raise ValueError(f"Malformed sub-header encountered: {lines[offset+1]}")
91 npoints = int(ms.group(1))
92 min_mass = float(ms.group(2))
93 max_mass = float(ms.group(3))
94 masses = np.linspace(min_mass, max_mass, npoints, endpoint=False)
95 branching_ratios = np.zeros(npoints)
96 # Now read the data lines (skipping the two header lines)
97 for line in lines[offset+2:offset+npoints+1]:
98 md = data_exp.match(line)
99 if md is None or len(md.groups()) != 2:
100 raise ValueError(f"Malformed data row encountered: {line}")
101 idx = int(md.group(1))
102 br = float(md.group(2))
103 branching_ratios[idx] = br
104 histograms[decay_code] = (masses, branching_ratios)
105 return histograms

107def make_interpolators(filepath, kind='linear'):
108 """
109 This function reads a file containing branching ratio histograms, and
110 returns a dictionary of interpolators of the branching ratios, indexed by
111 the decay string.
112 """
113 histogram_data = parse_histograms(filepath)
114 histograms = {}
115 for (hist_string, (masses, br)) in histogram_data.items():
116 histograms[hist_string] = scipy.interpolate.interp1d(
117 masses, br, kind=kind, bounds_error=False, fill_value=0, assume_sorted=True)
118 return histograms

120def get_br(histograms, channel, mass, couplings):
121 """
122 Utility function used to reliably query the branching ratio for a given
123 channel at a given mass, taking into account the correct coupling.
124 """
125 hist = histograms[channel['decay']]

177def fill_missing_channels(P8gen, max_total_br, decay_chains, epsilon=1e-6):
178 """
179 Add dummy channels for correct rejection sampling.
180
181 Even after rescaling the branching ratios, they do not sum up to unity
182 for most particles since we are ignoring SM processes.
183
184 This function adds a "filler" channel for each particle, in order to
185 preserve the ratios between different branching ratios.
186 """
187 top_level_particles = get_top_level_particles(decay_chains)
188 for particle in top_level_particles:
189 my_total_br = compute_total_br(particle, decay_chains)
190 remainder = 1 - my_total_br / max_total_br
191 assert(remainder > -epsilon)
192 assert(remainder < 1 + epsilon)
193 if remainder > epsilon:
194 add_dummy_channel(P8gen, particle, remainder)

196def add_dummy_channel(P8gen, particle, remainder):
197 """
198 Add a dummy channel to PYTHIA, with branching ratio equal to `remainder.`
199
200 The purpose of this function is to compensate for the absence of SM
201 channels, which are ignored when investigating rare processes. A dummy
202 decay channel is instead added to each particle in order to maintain the
203 correct ratios between the branching ratios of each particle to rare
204 processes. This is usually combined with a global reweighting of the
205 branching ratios.
206
207 In order to keep PYTHIA from complaining about charge conservation, a
208 suitable process is added which conserves electric charge.
209
210 All dummy channels can be identified by the presence of a photon among the
211 decay products.
212 """
213 pdg = P8gen.getPythiaInstance().particleData
214 charge = pdg.charge(particle)
215 if charge > 0:
216 P8gen.SetParameters(f'{particle}:addChannel 1 {remainder:.16} 0 22 -11')
217 elif charge < 0:
218 P8gen.SetParameters(f'{particle}:addChannel 1 {remainder:.16} 0 22 11')
219 else:
220 P8gen.SetParameters(f'{particle}:addChannel 1 {remainder:.16} 0 22 22')

222def compute_max_total_br(decay_chains):
223 """
224 This function computes the maximum total branching ratio for all decay chains.
225
226 In order to make the event generation as efficient as possible when
227 studying very rare processes, it is necessary to rescale the branching
228 ratios, while enforcing the invariant that any total branching ratio must
229 remain lower that unity.
230
231 This is accomplished by computing, for each particle, the total branching
232 ratio to processes of interest, and then dividing all branching ratios by
233 the highest of those.
234
235 Note: the decay chain length must be at most 2.
236 """
237 # For each top-level charmed particle, sum BR over all its decay chains
238 top_level_particles = get_top_level_particles(decay_chains)
239 total_branching_ratios = [compute_total_br(particle, decay_chains)
240 for particle in top_level_particles]
241 # Find the maximum total branching ratio
242 return max(total_branching_ratios)

244def compute_total_br(particle, decay_chains):
245 """
246 Returns the total branching ratio to HNLs for a given particle.
247 """
248 return sum(np.prod(branching_ratios)
249 for (top, branching_ratios) in decay_chains
250 if top == particle)
