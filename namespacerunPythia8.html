30def yBeam(Mproton = 0.938272081, pbeam = 400.):
31 Ebeam = ROOT.TMath.Sqrt(pbeam**2+Mproton**2)
32 betaCM = pbeam / (Ebeam + Mproton)
33 y_beam = 0.5*ROOT.TMath.Log( (1+betaCM)/(1-betaCM)) # https://arxiv.org/pdf/1604.02651.pdf
