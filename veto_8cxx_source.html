396 int veto::makeId(double z, double x, double y)
397 {
398  double Z = z / 10;
399  double r = sqrt(x * x + y * y);
400  double phi = 999;
401  if (y >= 0)
402  phi = acos(x / r);
403  else
404  phi = -acos(x / r) + 2 * TMath::Pi();
405 
406  phi = phi * 180 / TMath::Pi();
407  return (int)Z * 1000000 + (int)r * 1000 + (int)phi;

952 {

955  if (gMC->IsTrackEntering()) {

970 
971  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
972

977  TLorentzVector pos;
978  gMC->TrackPosition(pos);
979  TLorentzVector Mom;
980  gMC->TrackMomentum(Mom);

982  Double_t ymean = (fPos.Y() + pos.Y()) / 2.;
983  Double_t zmean = (fPos.Z() + pos.Z()) / 2.;
