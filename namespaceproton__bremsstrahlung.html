65def wba(p, theta, mDarkPhoton, epsilon):
66 """ Cross section weighting function in the Fermi-Weizsaeker-Williams approximation """
67 const = epsilon*epsilon*alphaQED / (2.*math.pi*H(p,theta,mDarkPhoton))
68
69 h2 = pow(H(p,theta,mDarkPhoton),2.)
70 oneMinusZSquare = pow(1.-zeta(p,theta),2.)
71 mp2 = mProton*mProton
72 mA2 = mDarkPhoton*mDarkPhoton
73
74 p1 = (1. + oneMinusZSquare) / zeta(p,theta)
75 p2 = ( 2. * zeta(p,theta) * (1.-zeta(p,theta)) * ( (2.*mp2 + mA2)/ H(p,theta,mDarkPhoton)
76 - pow(zeta(p,theta),2.)*2.*mp2*mp2/h2 ) )
