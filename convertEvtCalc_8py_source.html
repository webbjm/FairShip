74def convert_file(infile, outdir):
75 """
76 Create a ROOT file with a TTree containing the variables from the parsed input text file.

78 Parameters:
79 infile (str): Path to the input text file.
80 outdir (str): Name of the output directory, the filename will be the same

82 """
83 vars_names = [
84 "px_llp",
85 "py_llp",
86 "pz_llp",
87 "e_llp",
88 "mass_llp",
89 "pdg_llp",
90 "decay_prob",

92 "vy",
93 "vz",

97 "py_prod",
98 "pz_prod",
99 "e_prod",
100 "mass_prod",
101 "pdg_prod",
102 ]
103 fname = infile.split("/")[-1]
104 command = f"cp {infile} {outdir}/{fname}"
105
106 if os.path.isfile(f"{outdir}/{fname}"):
107 print(f"Warning: The file {outdir}/{fname} already exists.")
108 else:
109 os.system(command)
110
111 infile = f"{outdir}/{fname}"
112 parsed_data = parse_file(infile)
113 outfile = infile.split(".dat")[0] + ".root"
114 ncols = len(parsed_data[0][2])
115 nvardau = 6 # qualifiers for each daughter
116 remaining_vars = ncols - len(vars_names)
117
118 if (remaining_vars % nvardau)!=0:
119 raise ValueError(f"- convertEvtCalc - Error: number of daughters is not exact.")
120
121 ndau = remaining_vars // nvardau
122 print(f"- convertEvtCalc - Max multiplicity of daughters: {ndau}")
123
124 vars_names.extend(
125 f"{var}{i}" for i in range(1, ndau + 1) for var in daughter_vars
126 )
127
128 try:
129 check_consistency_infile(nvars=len(vars_names), ncols=ncols)
130 except ValueError as e:
131 print(f"- convertEvtCalc - Error: {e}")
132 sys.exit(1)
133
134 vars_names += ["ndau"]
135
136 if parsed_data:
137 root_file = r.TFile.Open(outfile, "RECREATE")
138 tree = r.TTree("LLP_tree", "LLP_tree")
139
140 branch_f = {}

142 branch_f[var] = np.zeros(1, dtype=float)
143 tree.Branch(var, branch_f[var], f"{var}/D")

145 for pt, sp, vars in parsed_data:
146 for row in zip(*vars):
147 for i, value in enumerate(row):
148 if i < len(vars_names)-1:
149 branch_f[vars_names[i]][0] = value
150 branch_f["ndau"][0] = ndau/1.
