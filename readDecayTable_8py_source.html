28def load(conffile = os.path.expandvars('$FAIRSHIP/python/DecaySelection.conf'), verbose=True):
29 f = open(conffile)
30 reader = csv.reader(f, delimiter=':')
31 configuredDecays = {}
32 for row in reader:
33 if not row: continue # skip empty lines
34 if str(row[0]).strip().startswith('#'):
35 continue # skip comment / header lines
36 channel = str(row[0]).strip()
37 flag = str(row[-1]).partition('#')[0].strip() # skip line comments
38 configuredDecays[channel] = flag
39 if verbose:
40 print('Activated decay channels (plus charge conjugates): ')
41 for channel in configuredDecays.keys():
42 if configuredDecays[channel] == 'yes':
43 print('\t'+channel)
44 return configuredDecays

46def addHNLdecayChannels(P8Gen, hnl, conffile=os.path.expandvars('$FAIRSHIP/python/DecaySelection.conf'), verbose=True):
47 """
48 Configures the HNL decay table in Pythia8
49 Inputs:
50 - P8Gen: an instance of ROOT.HNLPythia8Generator()
51 - hnl: an instance of hnl.HNL()
52 - conffile: a file listing the channels one wishes to activate
53 """
54 # First fetch the list of kinematically allowed decays
55 allowed = hnl.allowedChannels()
56 # Then fetch the list of desired channels to activate
57 wanted = load(conffile=conffile, verbose=verbose)
58 # Add decay channels
59 for dec in allowed:
60 if dec not in wanted:
61 print('addHNLdecayChannels ERROR: channel not configured!\t', dec)
62 quit()
63 if allowed[dec] == 'yes' and wanted[dec] == 'yes':
64 particles = [p for p in dec.replace('->',' ').split()]
65 children = particles[1:]
66 childrenCodes = [PDGcode(p) for p in children]
67 BR = hnl.findBranchingRatio(dec)
68 # Take care of Majorana modes
69 BR = BR/2.
70 codes = ' '.join([str(code) for code in childrenCodes])
71 P8Gen.SetParameters(f'9900015:addChannel = 1 {BR:.12} 0 {codes}')
72 # Charge conjugate modes
73 codes = ' '.join([(str(-1*code) if pdg.GetParticle(-code)!=None else str(code)) for code in childrenCodes])
74 P8Gen.SetParameters(f'9900015:addChannel = 1 {BR:.12} 0 {codes}')
75 # print "debug readdecay table",particles,children,BR
