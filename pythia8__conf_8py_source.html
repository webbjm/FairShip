129def configure(P8gen, mass, production_couplings, decay_couplings, process_selection,
130 deepCopy=False, debug=True):
131 """
132 This function configures a HNLPythia8Generator instance for SHiP usage.
133 """
134
135 if process_selection == True: # For backward compatibility
136 process_selection = 'inclusive'
137
138 # Wrap the Pythia8 object into a class logging all of its method calls
139 if debug:
140 pythia_log=open('pythia8_conf.txt','w')
141 P8gen = MethodLogger(P8gen, sink=pythia_log)
142
143 fairship_root = os.environ['FAIRSHIP']
144 histograms = make_interpolators(fairship_root + '/shipgen/branchingratios.dat')
145 P8gen.UseRandom3() # TRandom1 or TRandom3 ?
146 P8gen.SetMom(400) # beam momentum in GeV
147 if deepCopy: P8gen.UseDeepCopy()
148 pdg = ROOT.TDatabasePDG.Instance()
149 P8gen.SetParameters("Next:numberCount = 0")
150 # let strange particle decay in Geant4
151 make_particles_stable(P8gen, above_lifetime=1)
152
153 # Load particle & decay data
154 # ==========================
155
156 datafile = fairship_root + '/python/hnl_production.yaml'
157 with open(datafile) as f:
158 data = yaml.load(f, Loader=yaml.FullLoader)
159 all_channels = data['channels']
160
161 # Inclusive
162 # =========
163
164 if process_selection=='inclusive':
165 setup_pythia_inclusive(P8gen)
166
167 # Charm decays only (with secondary production from tau)
168 # ======================================================
169
170 if process_selection=='c':
171
172 selection = data['selections']['c']
173 for cmd in selection['parameters']:
174 P8gen.SetParameters(cmd)
175 add_hnl(P8gen, mass, decay_couplings)
176
177 # Add new charmed particles
178 # -------------------------
179
180 # Select all charmed particles (+ tau lepton)
181 c_particles = selection['particles']
182 tau_id = 15 # tau- Monte-Carlo ID
183 add_particles(P8gen, c_particles + [tau_id], data)
184
185 # Add HNL production channels from charmed particles
186 # --------------------------------------------------
187
188 # Find charm and tau decays to HNLs
189 c_channels = [ch for ch in all_channels if ch['id'] in c_particles]
190 tau_channels = [ch for ch in all_channels if ch['id'] == tau_id]
191 # Standard model process: tau+ production from D_s+ decay
192 ds_id = 431 # D_s+ Monte-Carlo ID
193 ds_tau_br = 0.0548 # SM branching ratio Br(D_s+ -> tau+ nu_tau) (source: PDG 2018)
194
195 # Compute the branching ratio scaling factor, taking into account
196 # secondary production from tau
197 # Decay chains are encoded as follows:
198 # [(top level id A, [br A -> B, br B -> C, ...]), ...]
199
200 # Most charm particles directly decay to HNLs
201 primary_decays = [(ch['id'], [get_br(histograms, ch, mass, production_couplings)])
202 for ch in c_channels]
203 # The D_s+ can indirectly produce a HNL by first producing a tau+
204 secondary_decays = [(ds_id, [ds_tau_br, get_br(histograms, ch, mass, production_couplings)])
205 for ch in tau_channels]
206 all_decays = primary_decays + secondary_decays
207
208 # Compute maximum total branching ratio (to rescale all BRs)
209 max_total_br = compute_max_total_br(all_decays)
210 exit_if_zero_br(max_total_br, process_selection, mass)
211 print_scale_factor(1/max_total_br)
212
213 # Add charm decays
214 for ch in c_channels:
215 add_channel(P8gen, ch, histograms, mass, production_couplings, 1/max_total_br)
216 # Add tau production from D_s+
217 # We can freely rescale Br(Ds -> tau) and Br(tau -> N X...) as long as
218 # Br(Ds -> tau -> N X...) remains the same.
219 # Here, we set Br(tau -> N) to unity to make event generation more efficient.
220 # The implicit assumption here is that we will disregard the tau during the analysis.
221 total_tau_br = sum(branching_ratios[1] for (_, branching_ratios) in secondary_decays)
222 assert(ds_tau_br*total_tau_br <= max_total_br + 1e-12)
223 P8gen.SetParameters("431:addChannel 1 {:.12} 0 -15 16"\
224 .format(ds_tau_br*total_tau_br/max_total_br))
225 # Add secondary HNL production from tau
226 for ch in tau_channels:
227 # Rescale branching ratios only if some are non-zero. Otherwise leave them at zero.
228 add_tau_channel(P8gen, ch, histograms, mass, production_couplings, 1/(total_tau_br or 1))
229
230 # Add dummy channels in place of SM processes
231 fill_missing_channels(P8gen, max_total_br, all_decays)
232
233 # List channels to confirm that Pythia has been properly set up
234 P8gen.List(9900015)
235
236 # B/Bc decays only
237 # ================
238
239 if process_selection in ['b', 'bc']:
240
241 selection = data['selections'][process_selection]
242 for cmd in selection['parameters']:
243 P8gen.SetParameters(cmd)
244 add_hnl(P8gen, mass, decay_couplings)
245
246 # Add particles
247 particles = selection['particles']
248 add_particles(P8gen, particles, data)
249
250 # Find all decay channels
251 channels = [ch for ch in all_channels if ch['id'] in particles]
252 decays = [(ch['id'], [get_br(histograms, ch, mass, production_couplings)]) for ch in channels]
253
254 # Compute scaling factor
255 max_total_br = compute_max_total_br(decays)
256 exit_if_zero_br(max_total_br, process_selection, mass)
257 print_scale_factor(1/max_total_br)
258
259 # Add beauty decays
260 for ch in channels:
261 add_channel(P8gen, ch, histograms, mass, production_couplings, 1/max_total_br)
262
263 # Add dummy channels in place of SM processes
264 fill_missing_channels(P8gen, max_total_br, decays)
265
266 P8gen.List(9900015)
267
268 if debug: pythia_log.close()
