116Bool_t strawtubes::ProcessHits(FairVolume* vol)
117{
119 //Set parameters at entrance of volume. Reset ELoss.
120 if ( gMC->IsTrackEntering() ) {
121 fELoss = 0.;
122 fTime = gMC->TrackTime() * 1.0e09;
123 fLength = gMC->TrackLength();
124 gMC->TrackPosition(fPos);
125 gMC->TrackMomentum(fMom);
126 }
127 // Sum energy loss for all steps in the active volume
128 fELoss += gMC->Edep();
129
130 // Create strawtubesPoint at exit of active volume
131 if ( gMC->IsTrackExiting() ||
132 gMC->IsTrackStop() ||
133 gMC->IsTrackDisappeared() ) {
134 if (fELoss == 0. ) { return kFALSE; }
135 TParticle* p=gMC->GetStack()->GetCurrentTrack();
136 Int_t pdgCode = p->GetPdgCode();
137 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
138 Int_t straw_uniqueId;
139 gMC->CurrentVolID(straw_uniqueId);
140 if (fVolumeID == straw_uniqueId) {
141 //std::cout << pdgCode<< " same volume again ? "<< straw_uniqueId << " exit:" << gMC->IsTrackExiting() << " stop:" << gMC->IsTrackStop() << " disappeared:" << gMC->IsTrackDisappeared()<< std::endl;
142 return kTRUE; }
143 fVolumeID = straw_uniqueId;
144 // # d = |pq . u x v|/|u x v|
145 TVector3 bot,top;
146 StrawEndPoints(straw_uniqueId,bot,top);
147 TLorentzVector Pos;
148 gMC->TrackPosition(Pos);
149 Double_t xmean = (fPos.X()+Pos.X())/2. ;
150 Double_t ymean = (fPos.Y()+Pos.Y())/2. ;
151 Double_t zmean = (fPos.Z()+Pos.Z())/2. ;
152 TVector3 pq = TVector3(top.x()-xmean,top.y()-ymean,top.z()-zmean );
153 TVector3 u = TVector3(bot.x()-top.x(),bot.y()-top.y(),bot.z()-top.z() );
154 TVector3 v = TVector3(fPos.X()-Pos.X(),fPos.Y()-Pos.Y(),fPos.Z()-Pos.Z());
155 TVector3 uCrossv = u.Cross(v);
156 Double_t dist2Wire = fabs(pq.Dot(uCrossv))/(uCrossv.Mag()+1E-8);
157 Double_t deltaTrackLength = gMC->TrackLength() - fLength;
158 AddHit(fTrackID, straw_uniqueId, TVector3(xmean, ymean, zmean),
159 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, deltaTrackLength,
160 fELoss,pdgCode,dist2Wire);
161 if (dist2Wire>fInner_Straw_diameter/2){
162 std::cout << "addhit " << dist2Wire<< " straw id " << straw_uniqueId << " pdgcode " << pdgCode<< " dot prod " << pq.Dot(uCrossv)<< std::endl;
163 std::cout << " exit:" << gMC->IsTrackExiting() << " stop:" << gMC->IsTrackStop() << " disappeared:" << gMC->IsTrackDisappeared()<< std::endl;
164 std::cout << " entry:" << fPos.X()<< " " << fPos.Y()<< " " << fPos.Z() << std::endl;
165 std::cout << " exit:" << Pos.X()<< " " << Pos.Y()<< " " << Pos.Z() << std::endl;
166 std::cout << " mean:" << xmean<< " " << ymean << " " << zmean << std::endl;
167 std::cout << " bot:" << bot.x()<< " " << bot.y() << " " << bot.z() << std::endl;
168 std::cout << " top:" << top.x()<< " " << top.y() << " " << top.z() << std::endl;
169 pq.Print();
170 u.Print();
171 v.Print();
172 uCrossv.Print();
173 }
174 // Increment number of strawtubes det points in TParticle
175 ShipStack* stack = (ShipStack*) gMC->GetStack();
176 stack->AddPoint(kStraw);
177 }
178 return kTRUE;
179}

334{
339 TGeoVolume *top = gGeoManager->GetTopVolume();
340 InitMedium("air");

342 InitMedium("ShipSens");
343 TGeoMedium *Se = gGeoManager->GetMedium("ShipSens");

356
357 gGeoManager->SetVisLevel(4);
358 gGeoManager->SetTopVisible();
359
360 //epsilon to avoid overlapping volumes
361 //Double_t eps=0.1;
362 Double_t eps=0.0001;
363 Double_t epsS=0.0001;
364 //width of frame
365 Double_t framewidth = 40.;
366 //width of view
367 Double_t viewwidth = fDeltaz_view-eps;
368 //width of plane
369 Double_t planewidth = fOuter_Straw_diameter+fDeltaz_layer12-eps;
370 //width of layer
371 Double_t layerwidth = fOuter_Straw_diameter;
372
373 Double_t rmin, rmax, dx, dy, dz, z, density,a,w;
374 Double_t par[20];
375 Int_t nel,numed,isvol,ifield;
376 Double_t radl, absl, TStationz;
377
378 Double_t yDim = (fStraws_per_layer+1) * fStraw_pitch /2. ; // put everything inside vacbox
379 //arguments of box are half-lengths;
380 TGeoBBox *detbox1 = new TGeoBBox("detbox1", fStraw_length+fFrame_lateral_width, ftr34ydim+fFrame_lateral_width, fDeltaz_frame/2.);
381 TGeoBBox *detbox2 = new TGeoBBox("detbox2", fStraw_length+eps, ftr34ydim+eps, fDeltaz_frame/2.+eps);
382
383 TGeoBBox *detbox1_12 = new TGeoBBox("detbox1_12", fStraw_length_12+fFrame_lateral_width, ftr12ydim+fFrame_lateral_width, fDeltaz_frame/2.);
384 TGeoBBox *detbox2_12 = new TGeoBBox("detbox2_12", fStraw_length_12+eps, ftr12ydim+eps, fDeltaz_frame/2.+eps);
385 TGeoCompositeShape *detcomp1 = new TGeoCompositeShape("detcomp1", "detbox1-detbox2");
386 TGeoCompositeShape *detcomp1_12 = new TGeoCompositeShape("detcomp1_12", "detbox1_12-detbox2_12");
387 TGeoBBox *vetovacbox;
388 TGeoCompositeShape *detcomp1_veto;
389 if (fStraw_length_veto>1){
390 TGeoBBox *detbox1_veto = new TGeoBBox("detbox1_veto", fStraw_length_veto+1., fvetoydim+1., fDeltaz_view/2.);
391 TGeoBBox *detbox2_veto = new TGeoBBox("detbox2_veto", fStraw_length_veto+eps, fvetoydim+eps, fDeltaz_view/2.+eps);
392
393 //the station sits inside a vacuum box
394 //TGeoBBox *vetovacbox = new TGeoBBox("Vetovacbox", fVacBox_x, fVacBox_y, fDeltaz_view );
395 vetovacbox = new TGeoBBox("Vetovacbox", fStraw_length_veto+75., fvetoydim+75., fDeltaz_view );
396 detcomp1_veto = new TGeoCompositeShape("detcomp1_veto", "detbox1_veto-detbox2_veto");
397 }
398 // Volume: straw
399 rmin = fInner_Straw_diameter/2.;
400 rmax = fOuter_Straw_diameter/2.;

415 straw_veto->SetVisibility(kTRUE);
416 }

418 rmin = fWire_thickness/2.+epsS;
419 rmax = fInner_Straw_diameter/2.-epsS;

482

494
495 for (Int_t pnb=0; pnb<1; pnb++) {
496 //plane loop
497 TString nmplane_veto = nmveto+"_plane_"; nmplane_veto += pnb;
498 //width of the planes: z distance between layers + outer straw diameter
499 TGeoBBox *plane_veto = new TGeoBBox("plane box", fStraw_length_veto+eps/2., fvetoydim+eps/2., planewidth/2.+3.*eps/2.);
500 TGeoVolume *planebox_veto = new TGeoVolume(nmplane_veto, plane_veto, med);
501 //the planebox sits in the viewframe
502 //hence z translate the plane wrt to the view
503
504 t5.SetTranslation(0, 0,(vnb-1./2.)*fDeltaz_view+(pnb-1./2.)*fDeltaz_plane12);
505 TGeoCombiTrans d5(t5, r5);
506 TGeoHMatrix *j5 = new TGeoHMatrix(d5);
507 vetovac->AddNode(planebox_veto, statnb*10000000+vnb*1000000+pnb*100000,j5);
508
509 for (Int_t lnb=0; lnb<2; lnb++) {
510 TString nmlayer_veto = nmplane_veto+"_layer_"; nmlayer_veto += lnb;
511 //width of the layer: (plane width-2eps)/2
512 layer_veto = new TGeoBBox("layer box_veto", fStraw_length_veto+eps/4., fvetoydim+eps/4., layerwidth/2.+eps/4.);
513 TGeoVolume *layerbox_veto = new TGeoVolume(nmlayer_veto, layer_veto, med);
514 //z translate the layerbox wrt the plane box (which is already rotated)
515 planebox_veto->AddNode(layerbox_veto, statnb*10000000+vnb*1000000+pnb*100000+lnb*10000,new TGeoTranslation(0,0,(lnb-1./2.)*fDeltaz_layer12));
516 //layer loop
517 TGeoRotation r6v;
518 TGeoTranslation t6v;
519 Int_t nr = statnb*10000000+vnb*1000000+pnb*100000+lnb*10000;
520 for (Int_t snb=1; snb<fStraws_per_layer_veto; snb++) {
521 //straw loop
522 t6v.SetTranslation(0,fvetoydim-fStraw_pitch*snb-fOffset_plane12*pnb+lnb*fOffset_layer12,0);
523 r6v.SetAngles(90,90,0);

526
527 layerbox_veto->AddNode(straw_veto,nr+1000+snb,h6v);
528 layerbox_veto->AddNode(gas_veto, nr+2000+snb,h6v);
529 layerbox_veto->AddNode(wire_veto, nr+3000+snb,h6v);
530 //end of straw loop
531 }
532 //end of layer loop
533 }
534 //end of plane loop
535 }
536 //end of view loop
537 } }
538 // end of veto station loop
539 //Tracking stations
540 //statnb=station number; vnb=view number; pnb=plane number; lnb=layer number; snb=straw number
541
542 //New scalable endpoints of vacuum boxes which cover rotated view frames
543
