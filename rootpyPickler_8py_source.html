6"""Pickle python data into a ROOT file, preserving references to ROOT objects.
7
8This module allows pickling python objects into a ROOT file. The python
9objects may contain references to named ROOT objects. If one has set up a
10structure of python objects to hold ROOT histograms, this provides a convenient
11way of saving and restoring your histograms. The pickled python data are
12stored in an additional string object in the ROOT file; any ROOT objects are
13stored as usual. (Thus, ROOT files written by the pickler can be read just
14like any other ROOT file if you don't care about the python data.)
15
16Here's an example of writing a pickle::
17
18from rootpyPickler import Pickler
19pkl=Pickler(file)
20pkl.dump(ShipGeo,'ShipGeo')
21
22from rootpyPickler import Unpickler
23upkl = Unpickler(f)
24ShipGeo = upkl.load('ShipGeo')
25
26The following additional notes apply:
27
28* Pickling may not always work correctly for the case of python objects
29 deriving from ROOT objects. It will probably also not work for the case of
30 ROOT objects which do not derive from TObject.
31
32* When the pickled data are being read, if a class doesn't exist,
33 a dummy class with no methods will be used instead. This is different
34 from the standard pickle behavior (where it would be an error), but it
35 simplifies usage in the common case where the class is being used to hold
36 histograms, and its methods are entirely concerned with filling the
37 histograms.
38
39* When restoring a reference to a ROOT object, the default behavior
40 is to not read the ROOT object itself, but instead to create a proxy. The
41 ROOT object will then be read the first time the proxy is accessed. This can
42 help significantly with time and memory usage if you're only accessing a
43 small fraction of the ROOT objects, but it does mean that you need to keep
44 the ROOT file open. Pass use_proxy=0 to disable this behavior.
45
46"""

117 self.__f = f
118 self.__pid = pid

121 def __getattr__(self, a):
122 if self.__o is None:

136class Pickler(pickle.Pickler):
137 def __init__(self, file, proto=0):
138 """Create a root pickler.
139 `file` should be a ROOT TFile. `proto` is the python pickle protocol
140 version to use. The python part will be pickled to a ROOT
141 TObjString called _pickle; it will contain references to the
142 ROOT objects.
143 """
144 self.__file = file
145 self.__keys = file.GetListOfKeys()
146 self.__io = IO_Wrapper()
147 self.__pmap = {}
148 super().__init__(self.__io, proto)
149
150 def dump(self, obj, key=None):
151 """Write a pickled representation of obj to the open TFile."""
152 if key is None:
153 key = '_pickle'
154 if 1>0:
155 self.__file.cd()
156 super().dump(obj)
157 s = ROOT.TObjString(self.__io.getvalue())
158 self.__io.reopen()
159 s.Write(key)
160 self.__file.GetFile().Flush()
161 self.__pmap.clear()
162
163 def clear_memo(self):
164 """Clears the pickler's internal memo."""
165 self.__pickle.memo.clear()

167 def persistent_id(self, obj):
168 if hasattr(obj, '_ROOT_Proxy__obj'):
169 obj = obj._ROOT_Proxy__obj()
170 if isinstance(obj, ROOT.TObject):
171 """
172 Write the object, and return the resulting NAME;CYCLE.
173 We used to do this::
174
175 o.Write()
176 k = self.__file.GetKey(o.GetName())
177 pid = "{0};{1:d}".format(k.GetName(), k.GetCycle())
178
179 It turns out, though, that destroying the python objects
180 referencing the TKeys is quite expensive (O(logN) where N is the
181 total number of pyroot objects?). Although we want to allow for
182 the case of saving multiple objects with the same name, the most
183 common case is that the name has not already been written to the
184 file. So we optimize for that case, doing the key lookup before we
185 write the object, not after. (Note further: GetKey() is very slow
186 if the key does not actually exist, as it does a linear search of
187 the key list. We use FindObject instead for the initial
188 lookup, which is a hashed lookup, but it is not guaranteed to
189 find the highest cycle. So if we do find an existing key, we
190 need to look up again using GetKey.
191 """
192 nm = obj.GetName()
193 key = self.__keys.FindObject(nm)
194 obj.Write()
195 if key:
196 key = self.__file.GetKey(nm)
197 pid = f'{nm};{key.GetCycle():d}'
198 else:
199 pid = nm + ';1'
200 return pid

204 def __init__(self, root_file, use_proxy=True, use_hash=False):
205 """Create a ROOT unpickler.
206 `file` should be a ROOT TFile.
207 """
208 global xserial
209 xserial += 1
210 self.__use_proxy = use_proxy
211 self.__file = root_file

213 self.__n = 0
214 self.__serial = f'{xserial:d}-'.encode()
215 xdict[self.__serial] = root_file
216 super().__init__(self.__io)

218 if use_hash:
219 htab = {}
220 ctab = {}
221 for k in root_file.GetListOfKeys():
222 nm = k.GetName()
223 cy = k.GetCycle()
224 htab[(nm, cy)] = k
225 if cy > ctab.get(nm, 0):
226 ctab[nm] = cy
227 htab[(nm, 9999)] = k
228 root_file._htab = htab

232 nm = nm0
233 ipos = nm.find(';')
234 if ipos >= 0:

236 if cy == '*':
237 cy = 10000

239 cy = int(cy)
240 nm = nm[:ipos - 1]
241 else:
242 cy = 9999
243 ret = htab.get((nm, cy), None)
244 if not ret:
245 print(("warning didn't find {0} {1} {2}",nm, cy, len(htab) ))
246 return oget(nm0)

249 #del ctx

253 def load(self, key=None):
254 """Read a pickled object representation from the open file."""
255 if key is None:
256 key = '_pickle'
257 obj = None
258 if _compat_hooks:
259 save = _compat_hooks[0]()
260 try:
261 self.__n += 1
262 s = self.__file.Get(key + f';{self.__n:d}')
263 self.__io.setvalue(s.GetName())
264 obj = super().load()
265 self.__io.reopen()
266 finally:
267 if _compat_hooks:
268 save = _compat_hooks[1](save)
269 return obj
