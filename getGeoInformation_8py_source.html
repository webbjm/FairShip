11def local2Global(n):

19 if options.moreInfo:
20 x = ROOT.gGeoManager.GetVerboseLevel()

24 ROOT.gGeoManager.SetVerboseLevel(x)

26 Info['locorign'] = o

28 globOrigin = array('d',[0,0,0])
29 nav.LocalToMaster(local,globOrigin)
30 Info['origin'] = globOrigin

32 [tmp.GetDX()+o[0],o[1],o[2]],
33 [o[0],-tmp.GetDY()+o[1],o[2]],
34 [o[0],tmp.GetDY()+o[1],o[2]],
35 [o[0],o[1],-tmp.GetDZ()+o[2]],[o[0],o[1],tmp.GetDZ()+o[2]]
36 ]
37 shifted = []
38 for s in shifts:
39 local = array('d',s)
40 glob = array('d',[0,0,0])
41 nav.LocalToMaster(local,glob)
42 shifted.append([glob[0],glob[1],glob[2]])
43 Info['boundingbox']={}
44 for j in range(3):
45 jmin = 1E30
46 jmax = -1E30
47 for s in shifted:
48 if s[j]<jmin: jmin = s[j]
49 if s[j]>jmax: jmax = s[j]
50 Info['boundingbox'][j]=[jmin,jmax]
51 return Info
