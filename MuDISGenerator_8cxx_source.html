49Double_t MuDISGenerator::MeanMaterialBudget(const Double_t *start, const Double_t *end, Double_t *mparam)
50{
51 //
52 // Calculate mean material budget and material properties between
53 // the points "start" and "end".
54 //
55 // "mparam" - parameters used for the energy and multiple scattering
56 // corrections:
57 //
58 // mparam[0] - mean density: sum(x_i*rho_i)/sum(x_i) [g/cm3]
59 // mparam[1] - equivalent rad length fraction: sum(x_i/X0_i) [adimensional]
60 // mparam[2] - mean A: sum(x_i*A_i)/sum(x_i) [adimensional]
61 // mparam[3] - mean Z: sum(x_i*Z_i)/sum(x_i) [adimensional]
62 // mparam[4] - length: sum(x_i) [cm]
63 // mparam[5] - Z/A mean: sum(x_i*Z_i/A_i)/sum(x_i) [adimensional]
64 // mparam[6] - number of boundary crosses
65 // mparam[7] - maximum density encountered (g/cm^3)
66 //
67 // Origin: Marian Ivanov, Marian.Ivanov@cern.ch
68 //
69 // Corrections and improvements by
70 // Andrea Dainese, Andrea.Dainese@lnl.infn.it,
71 // Andrei Gheata, Andrei.Gheata@cern.ch
72 //
73
74 mparam[0]=0; mparam[1]=1; mparam[2] =0; mparam[3] =0;
75 mparam[4]=0; mparam[5]=0; mparam[6]=0; mparam[7]=0;
76 //
77 Double_t bparam[6]; // total parameters
78 Double_t lparam[6]; // local parameters
79
80 for (Int_t i=0;i<6;i++) bparam[i]=0;
81
82 if (!gGeoManager) {
83 return 0.;
84 }
85 //
86 Double_t length;
87 Double_t dir[3];
88 length = TMath::Sqrt((end[0]-start[0])*(end[0]-start[0])+
89 (end[1]-start[1])*(end[1]-start[1])+
90 (end[2]-start[2])*(end[2]-start[2]));
91 mparam[4]=length;
92 if (length<TGeoShape::Tolerance()) return 0.0;
93 Double_t invlen = 1./length;
94 dir[0] = (end[0]-start[0])*invlen;
95 dir[1] = (end[1]-start[1])*invlen;
96 dir[2] = (end[2]-start[2])*invlen;
97
98 // Initialize start point and direction
99 TGeoNode *currentnode = 0;
100 TGeoNode *startnode = gGeoManager->InitTrack(start, dir);
101 if (!startnode) {
102 //AliErrorClass(Form("start point out of geometry: x %f, y %f, z %f",
103 // start[0],start[1],start[2]));
104 return 0.0;
105 }
106 TGeoMaterial *material = startnode->GetVolume()->GetMedium()->GetMaterial();
107 lparam[0] = material->GetDensity();
108 if (lparam[0] > mparam[7]) mparam[7]=lparam[0];
109 lparam[1] = material->GetRadLen();
110 lparam[2] = material->GetA();
111 lparam[3] = material->GetZ();
112 lparam[4] = length;
113 lparam[5] = lparam[3]/lparam[2];
114 if (material->IsMixture()) {
115 TGeoMixture * mixture = (TGeoMixture*)material;
116 lparam[5] =0;
117 Double_t sum =0;
118 for (Int_t iel=0;iel<mixture->GetNelements();iel++){
119 sum += mixture->GetWmixt()[iel];
120 lparam[5]+= mixture->GetZmixt()[iel]*mixture->GetWmixt()[iel]/mixture->GetAmixt()[iel];
121 }
122 lparam[5]/=sum;
123 }
124
125 // Locate next boundary within length without computing safety.
126 // Propagate either with length (if no boundary found) or just cross boundary
127 gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
128 Double_t step = 0.0; // Step made
129 Double_t snext = gGeoManager->GetStep();
130 // If no boundary within proposed length, return current density
131 if (!gGeoManager->IsOnBoundary()) {
132 mparam[0] = lparam[0];
133 mparam[1] = lparam[4]/lparam[1];
134 mparam[2] = lparam[2];
135 mparam[3] = lparam[3];
136 mparam[4] = lparam[4];
137 return lparam[0];
138 }
139 // Try to cross the boundary and see what is next
140 Int_t nzero = 0;
141 while (length>TGeoShape::Tolerance()) {
142 currentnode = gGeoManager->GetCurrentNode();
143 if (snext<2.*TGeoShape::Tolerance()) nzero++;
144 else nzero = 0;
145 if (nzero>3) {
146 // This means navigation has problems on one boundary
147 // Try to cross by making a small step
148 //AliErrorClass("Cannot cross boundary\n");
149 mparam[0] = bparam[0]/step;
150 mparam[1] = bparam[1];
151 mparam[2] = bparam[2]/step;
152 mparam[3] = bparam[3]/step;
153 mparam[5] = bparam[5]/step;
154 mparam[4] = step;
155 mparam[0] = 0.; // if crash of navigation take mean density 0
156 mparam[1] = 1000000; // and infinite rad length
157 return bparam[0]/step;
158 }
159 mparam[6]+=1.;
160 step += snext;
161 bparam[1] += snext/lparam[1];
162 bparam[2] += snext*lparam[2];
163 bparam[3] += snext*lparam[3];
164 bparam[5] += snext*lparam[5];
165 bparam[0] += snext*lparam[0];
166
167 if (snext>=length) break;
168 if (!currentnode) break;
169 length -= snext;
170 material = currentnode->GetVolume()->GetMedium()->GetMaterial();
171 lparam[0] = material->GetDensity();
172 if (lparam[0] > mparam[7]) mparam[7]=lparam[0];
173 lparam[1] = material->GetRadLen();
174 lparam[2] = material->GetA();
175 lparam[3] = material->GetZ();
176 lparam[5] = lparam[3]/lparam[2];
177 if (material->IsMixture()) {
178 TGeoMixture * mixture = (TGeoMixture*)material;
179 lparam[5]=0;
180 Double_t sum =0;
181 for (Int_t iel=0;iel<mixture->GetNelements();iel++){
182 sum+= mixture->GetWmixt()[iel];
183 lparam[5]+= mixture->GetZmixt()[iel]*mixture->GetWmixt()[iel]/mixture->GetAmixt()[iel];
184 }
185 lparam[5]/=sum;
186 }
187 gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
188 snext = gGeoManager->GetStep();
189 }
190 mparam[0] = bparam[0]/step;
191 mparam[1] = bparam[1];
192 mparam[2] = bparam[2]/step;
193 mparam[3] = bparam[3]/step;
194 mparam[5] = bparam[5]/step;
195 return bparam[0]/step;
196}
