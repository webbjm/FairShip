507 def extrapolateToPlane(fT,z):
508  # etrapolate to a plane perpendicular to beam direction (z)
509  rc,pos,mom = False,None,None
510  parallelToZ = ROOT.TVector3(0., 0., 1.)
511  fst = fT.getFitStatus()
512  if fst.isFitConverged():
513  # test for fit status for each point
514  if fT.getPoint(0).getFitterInfo() and fT.getPoint(1).getFitterInfo():
515  fstate0,fstate1 = fT.getFittedState(0),fT.getFittedState(1)
516  fPos0,fPos1 = fstate0.getPos(),fstate1.getPos()
517  if abs(z-fPos0.z()) < abs(z-fPos1.z()): fstate = fstate0
518  else: fstate = fstate1
519  zs = z
520  NewPosition = ROOT.TVector3(0., 0., zs)
521  rep = ROOT.genfit.RKTrackRep(13*cmp(fstate.getPDG(),0) )
522  state = ROOT.genfit.StateOnPlane(rep)
523  pos,mom = fstate.getPos(),fstate.getMom()
524  rep.setPosMom(state,pos,mom)
525  detPlane = ROOT.genfit.DetPlane(NewPosition, parallelToZ)
526  detPlanePtr = ROOT.genfit.SharedPlanePtr(detPlane)
527  try:
528  rep.extrapolateToPlane(state, detPlanePtr)
529  pos,mom = state.getPos(),state.getMom()
530  rc = True
531  except:
532  print('error with extrapolation')
533  pass
534  if not rc:
535  # use linear extrapolation
536  px,py,pz = mom.X(),mom.Y(),mom.Z()
537  lam = (z-pos.Z())/pz
538  pos = ROOT.TVector3( pos.X()+lam*px, pos.Y()+lam*py, z )
539  return rc,pos,mom

545 def decodeDetectorID(detID):
546  """
547  Decodes detector ID.
548  Parameters
549  ----------
550  detID : int or array-like
551  Detector ID values.
552  Returns
553  -------
554  statnb : int or array-like
555  Station numbers.
556  vnb : int or array-like
557  View numbers.
558  pnb : int or array-like
559  Plane numbers.
560  lnb : int or array-like
561  Layer numbers.
562  snb : int or array-like
563  Straw tube numbers.
564  """
565 
566  statnb = detID // 10000000
567  vnb = (detID - statnb * 10000000) // 1000000
568  pnb = (detID - statnb * 10000000 - vnb * 1000000) // 100000
569  lnb = (detID - statnb * 10000000 - vnb * 1000000 - pnb * 100000) // 10000
570  snb = detID - statnb * 10000000 - vnb * 1000000 - pnb * 100000 - lnb * 10000 - 2000
571 
572  return statnb, vnb, pnb, lnb, snb

576 def fracMCsame(trackids):
577  """
578  Estimates max fraction of true hit labels for a recognized track.
579  trackids : array_like
580  hit indexes of a recognized track
581  Retunrs
582  -------
583  frac : float
584  Max fraction of true hit labels.
585  tmax : int
586  True hit label with max fraction in a recognized track.
587  """
588 
589  track={}
590  nh=len(trackids)
591 
592  for tid in trackids:
593  if tid in track:
594  track[tid] += 1
595  else:
596  track[tid] = 1
597 
598  #now get track with largest number of hits
599  if track != {}:
600  tmax = max(track, key=track.get)
601  else:
602  track = {-999: 0}
603  tmax = -999
604 
605  frac=0.0
606  if nh > 0:
607  frac = float(track[tmax]) / float(nh)
608 
609  return frac, tmax
