6def expand_env(string):
7 """
8 Expand environment variables in string:
9 $HOME/bin -> /home/user/bin
10 """
11 while True:
12 m = re.search("(\${*(\w+)}*)", string)
13 if m is None:
14 break
15 (env_token, env_name) = m.groups()
16 assert env_name in os.environ, "Environment variable '%s' is not defined" % env_name
17 env_value = os.environ[env_name]
18 string = string.replace(env_token, env_value)
19 return string
