223  def Width_H_L(self, H, L):
224  """
225  Returns the RPV neutralino decay width into neutral meson and lepton
226 
227  Inputs:
228  - H is a string (name of the meson)
229  - L is a string (name of the lepton)
230  """
231  if self.MNMN < (mass(H)+mass(L)):
232  return 0.
233 
234  phsp=math.sqrt(self.MNMN**4+mass(H)**4+mass(L)**4-\
235  2*self.MNMN**2*mass(H)**2-2*self.MNMN**2*mass(L)**2-\
236  2*mass(H)**2*mass(L)**2)
237  tmp_width=0
238  if 'nu_mu' in L or 'nu_e' in L or 'nu_tau' in L:
239  tmp_width = phsp/(self.sfmasssfmass**4*128*u.pi*self.MNMN**3)*\
240  c.GST2['sneutrino']*c.decayConstant[H]**2*\
241  (self.MNMN**2+mass(L)**2-mass(H)**2)
242  else:
243  tmp_width = phsp/(self.sfmasssfmass**4*128*u.pi*self.MNMN**3)*\
244  c.GST2['slepton']*c.decayConstant[H]**2*\
245  (self.MNMN**2+mass(L)**2-mass(H)**2)
246 
247  if 'K*' in H or 'D*' in H or "phi" in H:
248  if 'nu_mu' in L or 'nu_e' in L or 'nu_tau' in L:
249  tmp_width = phsp/(self.sfmasssfmass**4*2*u.pi*self.MNMN**3)*\
250  c.GST2['tneutrino']*c.decayConstant[H]**2*\
251  (2*(self.MNMN**2-mass(L)**2)**2-mass(H)**2*(mass(H)**2+self.MNMN**2+mass(L)**2))
252  else:
253  tmp_width = phsp/(self.sfmasssfmass**4*2*u.pi*self.MNMN**3)*\
254  c.GST2['tlepton']*c.decayConstant[H]**2*\
255  (2*(self.MNMN**2-mass(L)**2)**2-mass(H)**2*(mass(H)**2+self.MNMN**2+mass(L)**2))
256  width=self.U2U2[1]*tmp_width
257  # contributions both from decay and production couplings
258  if self.benchbench==1 and ('K_' in H or 'K*' in H):
259  width=(self.U2U2[0]+self.U2U2[1])*tmp_width
260  # contributions only from production coupling
261  if self.benchbench==2 and ('eta' in H or 'phi' in H):
262  width=self.U2U2[0]*tmp_width
263 
264  # Majorana case (charge conjugate channels)
265  width = 2.*width
266  return width
