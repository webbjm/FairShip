39def PDGname(particle):
40 """
41 Trim particle name for use with the PDG database
42 """
43 if 'down' in particle: return 'd'
44 if 'up' in particle: return 'u'
45 if 'strange' in particle: return 's'
46 if 'charm' in particle: return 'c'
47 if 'bottom' in particle: return 'b'
48 if 'beauty' in particle: return 'b'
49 if 'top' in particle: return 't'
50 if '1' in particle:
51 particle = particle.replace('1',"'")
52 if (not (('-' in particle) or ('+' in particle) or ('0' in particle)
53 or ('nu_' in particle) or ('eta' in particle) or ('phi' in particle) )
54 and (particle not in ['d','u','s','c','b','t'])):
55 particle += '+'
56 return particle

85 """

89 return 4.1e-13

97 return 1.6e-12

223 def Width_H_L(self, H, L):
224 """
225 Returns the RPV neutralino decay width into neutral meson and lepton
226
227 Inputs:
228 - H is a string (name of the meson)
229 - L is a string (name of the lepton)
230 """
231 if self.MN < (mass(H)+mass(L)):
232 return 0.
233
234 phsp=math.sqrt(self.MN**4+mass(H)**4+mass(L)**4-\
235 2*self.MN**2*mass(H)**2-2*self.MN**2*mass(L)**2-\
236 2*mass(H)**2*mass(L)**2)
237 tmp_width=0
238 if 'nu_mu' in L or 'nu_e' in L or 'nu_tau' in L:
239 tmp_width = phsp/(self.sfmass**4*128*u.pi*self.MN**3)*\
240 c.GST2['sneutrino']*c.decayConstant[H]**2*\
241 (self.MN**2+mass(L)**2-mass(H)**2)
242 else:
243 tmp_width = phsp/(self.sfmass**4*128*u.pi*self.MN**3)*\
244 c.GST2['slepton']*c.decayConstant[H]**2*\
245 (self.MN**2+mass(L)**2-mass(H)**2)
246
247 if 'K*' in H or 'D*' in H or "phi" in H:
248 if 'nu_mu' in L or 'nu_e' in L or 'nu_tau' in L:
249 tmp_width = phsp/(self.sfmass**4*2*u.pi*self.MN**3)*\
250 c.GST2['tneutrino']*c.decayConstant[H]**2*\
251 (2*(self.MN**2-mass(L)**2)**2-mass(H)**2*(mass(H)**2+self.MN**2+mass(L)**2))
252 else:
253 tmp_width = phsp/(self.sfmass**4*2*u.pi*self.MN**3)*\
254 c.GST2['tlepton']*c.decayConstant[H]**2*\
255 (2*(self.MN**2-mass(L)**2)**2-mass(H)**2*(mass(H)**2+self.MN**2+mass(L)**2))
256 width=self.U2[1]*tmp_width
257 # contributions both from decay and production couplings
258 if self.bench==1 and ('K_' in H or 'K*' in H):
259 width=(self.U2[0]+self.U2[1])*tmp_width
260 # contributions only from production coupling
261 if self.bench==2 and ('eta' in H or 'phi' in H):
262 width=self.U2[0]*tmp_width
263
264 # Majorana case (charge conjugate channels)
265 width = 2.*width
266 return width

270 """
271 Returns the Meson decay width to RPV neutralino and lepton
272 Inputs:
273 - H is a string (name of the meson)
274 - L is a string (name of the lepton)
275 """
276
277 if mass(H) < (self.MN+mass(L)):
278 return 0.

280 2*self.MN**2*mass(H)**2-2*self.MN**2*mass(L)**2-\
281 2*mass(H)**2*mass(L)**2)

283 tmp_width=0
284 if 'nu_mu' in L or 'nu_e' in L or 'nu_tau' in L:
285 tmp_width = phsp/(self.sfmass**4*64*u.pi*mass(H)**3)*\
286 c.GST2['sneutrino']*c.decayConstant[H]**2*\
287 (mass(H)**2-self.MN**2-mass(L)**2)
288 else:

292
293 if 'K*' in H or 'D*' in H or "phi" in H:
294 if 'nu_mu' in L or 'nu_e' in L or 'nu_tau' in L:
295 tmp_width = phsp/(self.sfmass**4*3*u.pi*mass(H)**3)*\
296 c.GST2['tneutrino']*c.decayConstant[H]**2*\
297 (mass(H)**2*(mass(H)**2+self.MN**2+mass(L)**2-2*(self.MN**2-mass(L)**2)**2))

303 width=self.U2[0]*tmp_width

331 def findDecayBranchingRatio(self, decayString):
332 """
333 Returns the branching ratio of the selected SUSY RPV neutralino decay channel
334
335 Inputs:
336 - decayString is a string describing the decay, in the form 'N -> stuff1 ... stuffN'
337 """
338 had = re.search(r'->\ (.+?)\ ', decayString).group(1)
339 decaysplit = decayString.split(' ')
340 for split in decaysplit:
341 if split.find('mu')>-1 or split.find('e')>-1 or split.find('tau')>-1:
342 lep = split
343 if had == 'pi+' or had == 'pi-' or had == 'pi0':
344 print("findBranchingRatio() ERROR: Pions in final "\
345 "state have not been implemented, please choose "\
346 "a different decay mode of out...\n")
347 print(self.decays)
348 return -999
349
350 corrdecstring = 'N -> %s %s'%(had,lep)
351 listdecs = self.decays[self.bench]
352 gooddec = False
353 print("findBranchingRation() INFO: "\
354 "You have chosen the decay: '",\
355 corrdecstring)
356 for dec in listdecs:
357 if corrdecstring in dec:
358 gooddec = True
359 if gooddec is False:
360 print("findBranchingRation() ERROR: Badly "\
361 "formulated decay string, please choose "\
362 "one of the following\n")
363 print(self.decays)
364 return -999
365
366 br = 0.
367 totalwidth = self.NdecayWidth()
368 if totalwidth > 0.0:
369 br = self.Width_H_L(had,lep)/totalwidth
