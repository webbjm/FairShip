183Int_t UpstreamTagger::InitMedium(const char* name)
184{
185
186 static FairGeoLoader *geoLoad=FairGeoLoader::Instance();
187 static FairGeoInterface *geoFace=geoLoad->getGeoInterface();
188 static FairGeoMedia *media=geoFace->getMedia();
189 static FairGeoBuilder *geoBuild=geoLoad->getGeoBuilder();
190
191 FairGeoMedium *ShipMedium=media->getMedium(name);
192
193 if (!ShipMedium)
194 {
195 Fatal("InitMedium","Material %s not defined in media file.", name);
196 return -1111;
197 }
198 TGeoMedium* medium=gGeoManager->GetMedium(name);
199 if (medium!=NULL)
200 return ShipMedium->getMediumIndex();
201
202 return geoBuild->createMedium(ShipMedium);
203
204 return 0;
205}

209Bool_t UpstreamTagger::ProcessHits(FairVolume* vol)
210{
212 //Set parameters at entrance of volume. Reset ELoss.
213 if ( gMC->IsTrackEntering() ) {
214 fELoss = 0.;
215 fTime = gMC->TrackTime() * 1.0e09;
216 fLength = gMC->TrackLength();
217 gMC->TrackPosition(fPos);
218 gMC->TrackMomentum(fMom);
219 }
220
221 // Sum energy loss for all steps in the active volume
222 fELoss += gMC->Edep();
223
224 // Create vetoPoint at exit of active volume
225 if ( gMC->IsTrackExiting() ||
226 gMC->IsTrackStop() ||
227 gMC->IsTrackDisappeared() ) {
228 if (fELoss == 0. ) { return kFALSE; }
229
230 fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
231
232 Int_t uniqueId;
233 gMC->CurrentVolID(uniqueId);
234 if (uniqueId>1000000) //Solid scintillator case
235 {
236 Int_t vcpy;
237 gMC->CurrentVolOffID(1, vcpy);
238 if (vcpy==5) uniqueId+=4; //Copy of half
239 }
240
241 TParticle* p = gMC->GetStack()->GetCurrentTrack();
242 Int_t pdgCode = p->GetPdgCode();
243 TLorentzVector Pos;
244 gMC->TrackPosition(Pos);
245 TLorentzVector Mom;
246 gMC->TrackMomentum(Mom);
247 Double_t xmean = (fPos.X()+Pos.X())/2. ;
248 Double_t ymean = (fPos.Y()+Pos.Y())/2. ;
249 Double_t zmean = (fPos.Z()+Pos.Z())/2. ;
250
251 AddHit(fTrackID, uniqueId, TVector3(xmean, ymean, zmean),
252 TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
253 fELoss,pdgCode,TVector3(Pos.X(), Pos.Y(), Pos.Z()),
254 TVector3(Mom.Px(), Mom.Py(), Mom.Pz()) );
255
256 // Increment number of veto det points in TParticle
257 ShipStack* stack = (ShipStack*) gMC->GetStack();
258 stack->AddPoint(kUpstreamTagger);
259 }
260
261 return kTRUE;
262}
